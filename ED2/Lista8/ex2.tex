void heap_decrease_key(int V[], int size, int idx, int key) {
    if (key > V[idx]) {
        printf("Error: Expected key less than %d, got %d.\n", V[idx], key);
        return;
    }

    V[idx] = key;
    max_heapify(V, size, idx);
}
